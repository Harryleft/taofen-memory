```typescript
// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\InfiniteScrollLoader.tsx
interface InfiniteScrollLoaderProps {
  onLoadMore: () => Promise<void>;
  hasMore: boolean;
  isLoading: boolean;
  children?: React.ReactNode;
}

const InfiniteScrollLoader: React.FC<InfiniteScrollLoaderProps> = ({
  onLoadMore,
  hasMore,
  isLoading,
  children
}) => {
  const observerRef = useRef<IntersectionObserver>();
  const loadMoreRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (isLoading || !hasMore) return;

    observerRef.current = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          onLoadMore();
        }
      },
      { threshold: 0.1 }
    );

    if (loadMoreRef.current) {
      observerRef.current.observe(loadMoreRef.current);
    }

    return () => observerRef.current?.disconnect();
  }, [onLoadMore, hasMore, isLoading]);

  return (
    <>
      {children}
      {hasMore && (
        <div ref={loadMoreRef} className="infinite-scroll-trigger">
          {isLoading && (
            <div className="infinite-scroll-loading">
              <div className="loading-spinner"></div>
              <span>加载更多...</span>
            </div>
          )}
        </div>
      )}
    </>
  );
};
```

// 增强移动端手势支持
const useMobileGestures = () => {
  const [touchStart, setTouchStart] = useState(0);
  const [touchEnd, setTouchEnd] = useState(0);

  const handleTouchStart = (e: React.TouchEvent) => {
    setTouchStart(e.targetTouches[0].clientX);
  };

  const handleTouchMove = (e: React.TouchEvent) => {
    setTouchEnd(e.targetTouches[0].clientX);
  };

  const handleTouchEnd = (onSwipe?: (direction: 'left' | 'right') => void) => {
    if (!touchStart || !touchEnd) return;
    
    const diff = touchStart - touchEnd;
    const threshold = 50;
    
    if (diff > threshold) {
      onSwipe?.('left');
    } else if (diff < -threshold) {
      onSwipe?.('right');
    }
  };

  return {
    handleTouchStart,
    handleTouchMove,
    handleTouchEnd
  };
};

// 在主组件中使用
const gestures = useMobileGestures();

// 期数切换手势
const handleSwipe = useCallback((direction: 'left' | 'right') => {
  if (!selectedIssue || issues.length === 0) return;
  
  const currentIndex = issues.findIndex(issue => issue.manifest === selectedIssue.manifest);
  
  if (direction === 'left' && currentIndex < issues.length - 1) {
    handleIssueSelect(issues[currentIndex + 1]);
  } else if (direction === 'right' && currentIndex > 0) {
    handleIssueSelect(issues[currentIndex - 1]);
  }
}, [selectedIssue, issues, handleIssueSelect]);

// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\types\tree.ts
export interface TreeNode {
  id: string;
  title: string;
  type: 'category' | 'publication' | 'issue';
  level: number;
  children?: TreeNode[];
  metadata?: {
    issueCount?: number;
    lastUpdated?: string;
    manifest?: string;
    icon?: string;
  };
}

// 转换现有数据为树形结构
const convertToTreeStructure = (publications: PublicationItem[]): TreeNode[] => {
  // 按首字母分类的示例
  const categories: Record<string, TreeNode> = {};
  
  publications.forEach(pub => {
    const firstLetter = pub.title.charAt(0).toUpperCase();
    const categoryKey = `category-${firstLetter}`;
    
    if (!categories[categoryKey]) {
      categories[categoryKey] = {
        id: categoryKey,
        title: firstLetter,
        type: 'category',
        level: 0,
        children: [],
        metadata: {
          icon: '📁',
          issueCount: 0
        }
      };
    }
    
    categories[categoryKey].children?.push({
      id: pub.id,
      title: pub.title,
      type: 'publication',
      level: 1,
      metadata: {
        issueCount: pub.issueCount,
        lastUpdated: pub.lastUpdated,
        icon: '📰'
      }
    });
    
    if (categories[categoryKey].metadata) {
      categories[categoryKey].metadata.issueCount = 
        (categories[categoryKey].metadata.issueCount || 0) + pub.issueCount;
    }
  });
  
  return Object.values(categories).sort((a, b) => a.title.localeCompare(b.title));
};
```

// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\TreeView.tsx
interface TreeViewProps {
  nodes: TreeNode[];
  onSelect: (node: TreeNode) => void;
  selectedId?: string;
}

const TreeView: React.FC<TreeViewProps> = ({ nodes, onSelect, selectedId }) => {
  const [expandedNodes, setExpandedNodes] = useState<Set<string>>(new Set());

  const toggleNode = useCallback((nodeId: string) => {
    setExpandedNodes(prev => {
      const newSet = new Set(prev);
      if (newSet.has(nodeId)) {
        newSet.delete(nodeId);
      } else {
        newSet.add(nodeId);
      }
      return newSet;
    });
  }, []);

  const renderNode = (node: TreeNode) => {
    const isExpanded = expandedNodes.has(node.id);
    const isSelected = selectedId === node.id;
    const hasChildren = node.children && node.children.length > 0;

    return (
      <div key={node.id} className="tree-node">
        <div
          className={`tree-node-content ${isSelected ? 'selected' : ''}`}
          style={{ paddingLeft: `${node.level * 20 + 8}px` }}
          onClick={() => {
            if (hasChildren) {
              toggleNode(node.id);
            } else {
              onSelect(node);
            }
          }}
        >
          {hasChildren && (
            <span className="tree-toggle">
              {isExpanded ? '▼' : '▶'}
            </span>
          )}
          {!hasChildren && <span className="tree-spacer"></span>}
          <span className="tree-icon">{node.metadata?.icon || '📄'}</span>
          <span className="tree-title">{node.title}</span>
          {node.metadata?.issueCount && (
            <span className="tree-count">({node.metadata.issueCount})</span>
          )}
        </div>
        
        {hasChildren && isExpanded && (
          <div className="tree-children">
            {node.children?.map(child => renderNode(child))}
          </div>
        )}
      </div>
    );
  };

  return (
    <div className="tree-view">
      {nodes.map(node => renderNode(node))}
    </div>
  );
};

// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\EnhancedViewer.tsx
interface EnhancedViewerProps {
  manifestUrl: string;
  onPageChange?: (page: number) => void;
  onZoomChange?: (zoom: number) => void;
}

const EnhancedViewer: React.FC<EnhancedViewerProps> = ({
  manifestUrl,
  onPageChange,
  onZoomChange
}) => {
  const iframeRef = useRef<HTMLIFrameElement>(null);
  const [viewerState, setViewerState] = useState({
    currentPage: 1,
    totalPages: 1,
    zoomLevel: 1,
    isFullscreen: false
  });

  // 与iframe通信
  useEffect(() => {
    const handleMessage = (event: MessageEvent) => {
      if (event.data.type === 'viewer-state-change') {
        setViewerState(event.data.state);
      }
    };

    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, []);

  // 发送命令到iframe
  const sendCommand = useCallback((command: string, data?: any) => {
    if (iframeRef.current?.contentWindow) {
      iframeRef.current.contentWindow.postMessage({
        type: 'viewer-command',
        command,
        data
      }, '*');
    }
  }, []);

  // 工具栏功能
  const handlePageChange = useCallback((page: number) => {
    sendCommand('goToPage', page);
    onPageChange?.(page);
  }, [sendCommand, onPageChange]);

  const handleZoomChange = useCallback((zoom: number) => {
    sendCommand('setZoom', zoom);
    onZoomChange?.(zoom);
  }, [sendCommand, onZoomChange]);

  const toggleFullscreen = useCallback(() => {
    if (!document.fullscreenElement) {
      iframeRef.current?.requestFullscreen();
    } else {
      document.exitFullscreen();
    }
  }, []);

  return (
    <div className="enhanced-viewer">
      <div className="viewer-toolbar">
        <div className="viewer-controls">
          <button
            onClick={() => handlePageChange(viewerState.currentPage - 1)}
            disabled={viewerState.currentPage <= 1}
          >
            ← 上一页
          </button>
          <span className="page-info">
            {viewerState.currentPage} / {viewerState.totalPages}
          </span>
          <button
            onClick={() => handlePageChange(viewerState.currentPage + 1)}
            disabled={viewerState.currentPage >= viewerState.totalPages}
          >
            下一页 →
          </button>
        </div>
        
        <div className="viewer-tools">
          <button onClick={() => handleZoomChange(viewerState.zoomLevel - 0.1)}>
            🔍-
          </button>
          <span className="zoom-level">
            {Math.round(viewerState.zoomLevel * 100)}%
          </span>
          <button onClick={() => handleZoomChange(viewerState.zoomLevel + 0.1)}>
            🔍+
          </button>
          <button onClick={toggleFullscreen}>
            {viewerState.isFullscreen ? '退出全屏' : '全屏'}
          </button>
        </div>
      </div>
      
      <iframe
        ref={iframeRef}
        src={manifestUrl}
        className="viewer-iframe"
        title="报刊查看器"
      />
    </div>
  );
};

// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\SearchFilter.tsx
interface SearchFilterProps {
  publications: PublicationItem[];
  onFilter: (filtered: PublicationItem[]) => void;
}

const SearchFilter: React.FC<SearchFilterProps> = ({ publications, onFilter }) => {
  const [searchTerm, setSearchTerm] = useState('');
  const [sortBy, setSortBy] = useState<'name' | 'date' | 'count'>('name');

  useEffect(() => {
    let filtered = [...publications];

    // 搜索过滤
    if (searchTerm.trim()) {
      const term = searchTerm.toLowerCase();
      filtered = filtered.filter(pub =>
        pub.title.toLowerCase().includes(term) ||
        pub.name.toLowerCase().includes(term)
      );
    }

    // 排序
    filtered.sort((a, b) => {
      switch (sortBy) {
        case 'name':
          return a.title.localeCompare(b.title, 'zh-CN');
        case 'date':
          if (!a.lastUpdated) return 1;
          if (!b.lastUpdated) return -1;
          return a.lastUpdated.localeCompare(b.lastUpdated, 'zh-CN');
        case 'count':
          return b.issueCount - a.issueCount;
        default:
          return 0;
      }
    });

    onFilter(filtered);
  }, [publications, searchTerm, sortBy, onFilter]);

  return (
    <div className="search-filter">
      <div className="search-input">
        <input
          type="text"
          placeholder="搜索报刊..."
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
        />
        <span className="search-icon">🔍</span>
      </div>
      
      <div className="filter-controls">
        <select
          value={sortBy}
          onChange={(e) => setSortBy(e.target.value as any)}
        >
          <option value="name">按名称排序</option>
          <option value="date">按日期排序</option>
          <option value="count">按期数排序</option>
        </select>
      </div>
    </div>
  );
};

// S:\vibe_coding\taofen_web\frontend\src\components\newspapers\useUserPreferences.ts
interface UserPreferences {
  defaultView: 'list' | 'grid';
  itemsPerPage: number;
  autoOpenSidebar: boolean;
  rememberLastPosition: boolean;
  theme: 'light' | 'dark';
}

const defaultPreferences: UserPreferences = {
  defaultView: 'list',
  itemsPerPage: 20,
  autoOpenSidebar: true,
  rememberLastPosition: true,
  theme: 'light'
};

const useUserPreferences = () => {
  const [preferences, setPreferences] = useState<UserPreferences>(() => {
    try {
      const saved = localStorage.getItem('newspapers-preferences');
      return saved ? { ...defaultPreferences, ...JSON.parse(saved) } : defaultPreferences;
    } catch {
      return defaultPreferences;
    }
  });

  const updatePreferences = useCallback((updates: Partial<UserPreferences>) => {
    const newPreferences = { ...preferences, ...updates };
    setPreferences(newPreferences);
    
    try {
      localStorage.setItem('newspapers-preferences', JSON.stringify(newPreferences));
    } catch (error) {
      console.error('保存偏好设置失败:', error);
    }
  }, [preferences]);

  return { preferences, updatePreferences };
};
```
